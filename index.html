Ungleicher Konsum - Ungleiche Emissionen

Die Klimakrise ist auch eine Frage der Verteilung: In Österreich verursacht das reichste Einkommenszehntel viermal so viele CO₂-Emissionen wie das ärmste. Während Ausgaben für Energie bei allen ähnlich sind, klaffen bei Mobilität und Güterkonsum die Emissionen massiv auseinander. Gleichzeitig würden ärmere Haushalte bei einer CO₂-Bepreisung überproportional belastet – sie geben einen größeren Teil ihres Einkommens für klimaschädlichen Konsum aus, ohne die finanziellen Mittel für Alternativen. Die Isotype-Darstellung macht diese Ungleichheit auf einen Blick erfassbar.

makeIsotype = function({
  inputData,
  shapeDomain = ["Lebensmittel", "Wohnen", "Energie", "Mobilität", "Güter", "Dienstleistungen"],
  shapeRange = [path1, path2, path3, path4, path5, path6],
  colorRange = colors_burger,
  xDomain = [0, 25],
  height = 300,
  size = 5,
  unit = "€",
  divisor = 1000,
  minExtent = 100,
  spacing = 10,
  plotWidth = null
}) {

  return vl.layer(
    vl.markPoint({
      size: size,
      stroke: "black",
      strokeOpacity: 1,
      strokeWidth: 0.75,
      opacity: 1,
      clip: false,
    })
    .encode(
      vl.y()
        .fieldN("name")
        .sort(shapeDomain)
        .axis({
          domain: false,
          ticks: false,
          title: null,
          labelOffset: 14,
          minExtent: minExtent,
        }),
      vl.x()
        .fieldQ("positions")
        .scale({
          domain: xDomain,
          nice: false,
          zero: false,
        })
        .axis({
          title: null,
          labels: false,
          ticks: false,
          domain: false,
          grid: false,
        }),
      vl.fill()
        .fieldN("name")
        .scale({
          range: colorRange,
        })
        .legend(false),
      vl.shape()
        .fieldN("name")
        .scale({
          domain: shapeDomain,
          range: shapeRange
        })
        .legend(false)
    ),
    vl.markText({
      align: "left",
      dx: 30,
      dy: 14,
      fontSize: 12,
      clip: false
    })
    .transform(
      vl.filter(`datum.positions === round(datum.value / ${divisor}) - 1`),
      vl.calculate(`format(datum.value, ',') + ' ${unit}'`).as("valueWithUnit")
    )
    .encode(
      vl.y()
        .fieldN("name"),
      vl.x()
        .fieldQ("positions"),
      vl.text()
        .fieldN("valueWithUnit"),
    )
  )
  .transform(
    vl.calculate(`sequence(0, round(datum.value / ${divisor}))`).as("positions"),
    vl.flatten(["positions"]),
  )
  .width(plotWidth !== null ? plotWidth : makeWidth(width))
  .height(height)
  .facet({
    column: {
      field: "Quantile",
      type: "nominal",
      title: null,
      sort: ["Arm", "Reich"],
      header: {
        labelFontSize: 15,
        labelOrient: "top",
        labelPadding: -6,
      },
    },
    spacing: spacing
  })
  .resolve({scale: {x: "shared"}})
  .data(inputData)
  .config(config);

}
